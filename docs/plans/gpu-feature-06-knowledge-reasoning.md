## 数据库设计

```sql
-- 因果关系表
CREATE TABLE causal_relations (
    id SERIAL PRIMARY KEY,
    cause_entity_id INTEGER,
    effect_entity_id INTEGER,
    cause_text TEXT,
    effect_text TEXT,
    confidence FLOAT,
    evidence TEXT,
    source_id INTEGER,
    created_at TIMESTAMP DEFAULT NOW()
);

-- 假设表
CREATE TABLE hypotheses (
    id SERIAL PRIMARY KEY,
    statement TEXT,
    rationale TEXT,
    suggested_experiment TEXT,
    confidence FLOAT,
    priority VARCHAR(10),
    status VARCHAR(20),  -- proposed/validated/rejected
    supporting_entities JSONB,
    evidence JSONB,
    created_by INTEGER REFERENCES members(id),
    created_at TIMESTAMP DEFAULT NOW(),
    updated_at TIMESTAMP DEFAULT NOW()
);

-- 冲突记录表
CREATE TABLE knowledge_conflicts (
    id SERIAL PRIMARY KEY,
    conflict_type VARCHAR(50),
    description TEXT,
    entity_ids INTEGER[],
    knowledge_ids INTEGER[],
    severity VARCHAR(20),  -- low/medium/high/critical
    status VARCHAR(20),  -- detected/analyzing/resolved
    resolution TEXT,
    created_at TIMESTAMP DEFAULT NOW(),
    resolved_at TIMESTAMP
);

-- 趋势预测表
CREATE TABLE trend_predictions (
    id SERIAL PRIMARY KEY,
    topic VARCHAR(200),
    prediction_type VARCHAR(50),
    prediction JSONB,
    confidence FLOAT,
    time_horizon VARCHAR(20),
    created_at TIMESTAMP DEFAULT NOW()
);
```
